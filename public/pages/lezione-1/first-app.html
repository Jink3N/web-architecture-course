🚀 Prima App Node.js

Creiamo la nostra prima applicazione Node.js completa step-by-step.

📁 Struttura Progetto

🏗️ Setup Iniziale

# Crea directory progetto
mkdir my-first-node-app
cd my-first-node-app

# Inizializza npm
npm init -y

# Struttura cartelle
mkdir src
mkdir public
mkdir routes

📦 package.json

{
  "name": "my-first-node-app",
  "version": "1.0.0",
  "description": "La mia prima app Node.js",
  "main": "src/app.js",
  "scripts": {
    "start": "node src/app.js",
    "dev": "nodemon src/app.js"
  },
  "dependencies": {
    "express": "^4.18.2"
  },
  "devDependencies": {
    "nodemon": "^3.0.1"
  }
}

💻 Codice Applicazione

🌐 src/app.js - Server Express

const express = require('express');
const path = require('path');

const app = express();
const PORT = process.env.PORT || 3000;

// Middleware
app.use(express.static('public'));
app.use(express.json());

// Routes
app.get('/', (req, res) => {
  res.sendFile(path.join(__dirname, '../public', 'index.html'));
});

app.get('/api/hello', (req, res) => {
  res.json({ 
    message: 'Hello from Node.js!',
    timestamp: new Date().toISOString()
  });
});

app.listen(PORT, () => {
  console.log(`🚀 Server running on http://localhost:${PORT}`);
});

🎨 public/index.html

<!DOCTYPE html>
<html>
<head>
    <title>My First Node.js App</title>
</head>
<body>
    <h1>🚀 Benvenuto in Node.js!</h1>
    <button onclick="testAPI()">Test API</button>
    <div id="result"></div>

    <script>
        async function testAPI() {
            const response = await fetch('/api/hello');
            const data = await response.json();
            document.getElementById('result').innerHTML = 
                `<p>${data.message}</p><p>${data.timestamp}</p>`;
        }
    </script>
</body>
</html>

🚀 Avvio Applicazione

Installa dipendenze: npm install

Avvia server: npm start

Sviluppo: npm run dev (con auto-reload)

Test: Apri http://localhost:3000
